<conditionalStmt> -> TK_SQL     TK_ID   TK_CALL         TK_WHILE        TK_ENDWHILE     TK_IF   TK_ELSE         TK_ENDIF        TK_READ         TK_WRITE        TK_RETURN 
<ioStmt> -> TK_SQL      TK_ID   TK_CALL         TK_WHILE        TK_ENDWHILE     TK_IF   TK_ELSE         TK_ENDIF        TK_READ         TK_WRITE        TK_RETURN
<funCallStmt> -> TK_SQL         TK_ID   TK_CALL         TK_WHILE        TK_ENDWHILE     TK_IF   TK_ELSE         TK_ENDIF        TK_READ         TK_WRITE        TK_RETURN

<moreExpansions> -> TK_SEM      TK_ASSIGNOP     TK_CL   TK_MUL  TK_DIV  TK_PLUS         TK_MINUS        TK_LT   TK_LE   TK_EQ   TK_GT   TK_GE   TK_NE

<inputParameters> -> TK_SEM
<idList> -> TK_SQR
<booleanExpression> -> TK_CL
